query1 = '''
SELECT *
FROM(SELECT 
        SE_NAME,
       AVG(TS_TIME_0_6_AMT) TS_TIME_0_6_AMT,
       AVG(TS_TIME_6_11_AMT) TS_TIME_6_11_AMT,
       AVG(TS_TIME_11_14_AMT) TS_TIME_11_14_AMT,
       AVG(TS_TIME_14_17_AMT) TS_TIME_14_17_AMT,
       AVG(TS_TIME_17_21_AMT) TS_TIME_17_21_AMT,
       AVG(TS_TIME_21_24_AMT) TS_TIME_21_24_AMT,
       (AVG(TS_TIME_0_6_AMT)+AVG(TS_TIME_6_11_AMT)+AVG(TS_TIME_11_14_AMT)+AVG(TS_TIME_14_17_AMT)+AVG(TS_TIME_17_21_AMT)+AVG(TS_TIME_21_24_AMT))/6 SUMAVG
       
FROM TIME_AMT_ANALYSIS T1, 
     ASTORE T2, 
     ASERVICE T3 
WHERE (T1.S_CODE = T2.S_CODE)
      and (T1.SE_CODE = T3.SE_CODE)
      and T1.TS_QUARTER = 1
      
Group By  SE_NAME
Order By SUMAVG DESC) a1
WHERE ROWNUM BETWEEN 1 AND 5
'''
